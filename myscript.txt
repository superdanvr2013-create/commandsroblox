copyPos.MouseButton1Click:Connect(function()
    local char = game.Players.LocalPlayer.Character
    if char and char:FindFirstChild("HumanoidRootPart") then
        pos1 = char.HumanoidRootPart.CFrame
        local p = char.HumanoidRootPart.Position
        posLabel.Text = string.format("%.1f, %.1f, %.1f", p.X, p.Y, p.Z)
    end
end)

-- Логика телепортации

    local char = game.Players.LocalPlayer.Character
    if char and char:FindFirstChild("HumanoidRootPart") and pos1 then
        char.HumanoidRootPart.CFrame = pos1
    end
end)
